Video with external controls

HTML5 video element with external JavaScript controls.


document.getElementById('player').addEventListener('timeupdate',function(e) {
	document.getElementById('out').innerHTML = Math.round(e.target.currentTime*100)/100;
}, true);
document.getElementById('play').addEventListener('click',function() {
	document.getElementById('player').play();
	return false;
}, true);
document.getElementById('pause').addEventListener('click',function() {
	document.getElementById('player').pause();
	return false;
}, true);
document.getElementById('volume').addEventListener('change',function(e) {
	document.getElementById('player').volume=e.target.value;
	return false;
}
